\tikzset{
    block/.style={rectangle, draw, thick, text centered, font=\bfseries, inner sep=.4cm}
}

    \node (tokenization) [block] {Tokenization};

    \draw[->] (embedding.east) -- ++(3, 0) node[midway, above] {Embeddings};
\end{tikzpicture}
